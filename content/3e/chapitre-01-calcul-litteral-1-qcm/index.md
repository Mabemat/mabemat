{{< qcm question="Développer $(x+3)(x-2)$ donne :" correct="0" >}}
- $x^2 + x - 6$
- $x^2 - x - 6$
- $x^2 + 6x - 6$
- $x^2 - 6$
{{< /qcm >}}

{{< qcm question="Factoriser $x^2 - 9$ donne :" correct="2" >}}
- $(x+9)(x-1)$
- $x(x-9)$
- $(x+3)(x-3)$
- $(x-3)^2$
{{< /qcm >}}

{{< qcm question="L'identité remarquable $(a+b)^2$ vaut :" correct="1" >}}
- $a^2 + b^2$
- $a^2 + 2ab + b^2$
- $a^2 - 2ab + b^2$
- $2a + 2b$
{{< /qcm >}}

{{< qcm question="Développer $(2x - 1)^2$ donne :" correct="0" >}}
- $4x^2 - 4x + 1$
- $4x^2 + 1$
- $4x^2 - 1$
- $2x^2 - 2x + 1$
{{< /qcm >}}

{{< qcm question="Factoriser $6x^2 + 4x$ donne :" correct="1" >}}
- $2(3x + 2)$
- $2x(3x + 2)$
- $2x(3x + 4)$
- $x(6x + 4)$
{{< /qcm >}}
