{{< qcm question="Factoriser $x^2 - 4x + 4$ donne :" correct="2" >}}
- $(x-2)(x+2)$
- $(x-4)(x+1)$
- $(x-2)^2$
- $(x+2)^2$
{{< /qcm >}}

{{< qcm question="Développer $(x-5)(x+5)$ donne :" correct="0" >}}
- $x^2 - 25$
- $x^2 + 25$
- $x^2 - 10x + 25$
- $x^2 + 10x - 25$
{{< /qcm >}}

{{< qcm question="Résoudre $x^2 = 9$ :" correct="1" >}}
- $x = 3$
- $x = 3$ ou $x = -3$
- $x = -3$
- $x = \pm 9$
{{< /qcm >}}

{{< qcm question="Factoriser $4x^2 - 1$ :" correct="2" >}}
- $(2x)^2 - 1$
- $2x(2x-1)$
- $(2x-1)(2x+1)$
- $(4x-1)(x+1)$
{{< /qcm >}}

{{< qcm question="$a^2 - b^2 = $" correct="1" >}}
- $(a-b)^2$
- $(a+b)(a-b)$
- $(a-b)(a+b)^2$
- $a^2 + b^2$
{{< /qcm >}}
