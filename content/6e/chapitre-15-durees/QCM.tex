\begin{Correction}
\begin{EXO}{}{}

 On ajoute les heures avec les heures et les minutes avec les minutes.\\$(2\text{ h }+1\text{ h })+(15\text{ min }+25\text{ min })={\color[HTML]{F15929}\boldsymbol{3\textbf{ h }40\textbf{ min}}}$.
\end{EXO}
