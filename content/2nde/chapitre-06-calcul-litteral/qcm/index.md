{{< qcm question="$(x+3)^2$ développé donne :" correct="1" >}}
- $x^2+9$
- $x^2+6x+9$
- $x^2+3x+9$
- $x^2+6x+6$
{{< /qcm >}}

{{< qcm question="$(2x-1)(2x+1)$ vaut :" correct="0" >}}
- $4x^2-1$
- $4x^2+1$
- $4x^2-4x+1$
- $4x^2-2$
{{< /qcm >}}

{{< qcm question="Factoriser $x^2-9$ donne :" correct="2" >}}
- $(x-3)^2$
- $(x+3)^2$
- $(x-3)(x+3)$
- $(x+9)(x-1)$
{{< /qcm >}}

{{< qcm question="$(a-b)^2$ développé donne :" correct="2" >}}
- $a^2-b^2$
- $a^2+2ab+b^2$
- $a^2-2ab+b^2$
- $a^2-ab+b^2$
{{< /qcm >}}

{{< qcm question="Factoriser $4x^2-12x+9$ donne :" correct="0" >}}
- $(2x-3)^2$
- $(2x+3)^2$
- $(4x-3)(x-3)$
- $(2x-9)^2$
{{< /qcm >}}

{{< qcm question="Résoudre $(x+2)(x-5) = 0$ :" correct="1" >}}
- $x = 2$ ou $x = -5$
- $x = -2$ ou $x = 5$
- $x = 10$
- $x = -10$
{{< /qcm >}}
